# 7.6 Implementing Custom Event Accessors

## Description

Custom event accessors allow you to implement custom logic for adding and removing event handlers. This provides fine-grained control over event subscription and unsubscription, enabling features like event filtering, logging, validation, and custom storage mechanisms.

### Key Concepts:
- **Custom Accessors**: Implement custom `add` and `remove` accessors for events
- **Event Storage**: Custom storage mechanisms for event handlers
- **Event Filtering**: Filter event subscriptions based on criteria
- **Event Logging**: Log event subscription and unsubscription
- **Event Validation**: Validate event handlers before adding them

### Common Patterns:
- **Event Filtering**: Only allow certain types of handlers
- **Event Logging**: Track event subscriptions for debugging
- **Event Validation**: Validate handlers before adding them
- **Custom Storage**: Store handlers in custom data structures
- **Event Chaining**: Chain events through custom logic

## Code Examples

### Example 1: Basic Custom Event Accessors

```csharp
using System;
using System.Collections.Generic;

// Class with custom event accessors
class CustomEventPublisher
{
    private EventHandler<string> _dataChanged;
    private readonly object _lockObject = new object();

    // Custom event accessors
    public event EventHandler<string> DataChanged
    {
        add
        {
            lock (_lockObject)
            {
                Console.WriteLine($"Adding handler: {value?.Method.Name}");
                _dataChanged += value;
            }
        }
        remove
        {
            lock (_lockObject)
            {
                Console.WriteLine($"Removing handler: {value?.Method.Name}");
                _dataChanged -= value;
            }
        }
    }

    public void ChangeData(string newData)
    {
        Console.WriteLine($"Changing data to: {newData}");
        _dataChanged?.Invoke(this, newData);
    }
}

// Event handler
class EventHandler
{
    private string _name;

    public EventHandler(string name)
    {
        _name = name;
    }

    public void OnDataChanged(object sender, string data)
    {
        Console.WriteLine($"  [{_name}] Data changed: {data}");
    }
}

class Program
{
    static void Main()
    {
        CustomEventPublisher publisher = new CustomEventPublisher();
        EventHandler handler1 = new EventHandler("Handler1");
        EventHandler handler2 = new EventHandler("Handler2");

        Console.WriteLine("=== Custom Event Accessors ===");
        
        // Subscribe to events (will trigger custom add accessor)
        publisher.DataChanged += handler1.OnDataChanged;
        publisher.DataChanged += handler2.OnDataChanged;

        // Trigger event
        publisher.ChangeData("New data");

        Console.WriteLine("\n=== Unsubscription ===");
        
        // Unsubscribe from events (will trigger custom remove accessor)
        publisher.DataChanged -= handler1.OnDataChanged;
        publisher.DataChanged -= handler2.OnDataChanged;

        // Trigger event again
        publisher.ChangeData("Another data");
    }
}
```

**Output:**
```

Adding handler: OnDataChanged
Adding handler: OnDataChanged
Changing data to: New data
  [Handler1] Data changed: New data
  [Handler2] Data changed: New data

=== Unsubscription ===
Removing handler: OnDataChanged
Removing handler: OnDataChanged
Changing data to: Another data
```

### Example 2: Event Filtering with Custom Accessors

```csharp
using System;
using System.Collections.Generic;

// Event handler interface for filtering
interface IEventHandler
{
    string HandlerName { get; }
    bool IsEnabled { get; }
}

// Custom event handler with filtering
class FilteredEventHandler : IEventHandler
{
    public string HandlerName { get; }
    public bool IsEnabled { get; set; }

    public FilteredEventHandler(string name, bool enabled = true)
    {
        HandlerName = name;
        IsEnabled = enabled;
    }

    public void OnDataChanged(object sender, string data)
    {
        if (IsEnabled)
        {
            Console.WriteLine($"  [{HandlerName}] Data changed: {data}");
        }
    }
}

// Class with filtered event accessors
class FilteredEventPublisher
{
    private EventHandler<string> _dataChanged;
    private readonly List<IEventHandler> _registeredHandlers = new List<IEventHandler>();
    private readonly object _lockObject = new object();

    // Custom event accessors with filtering
    public event EventHandler<string> DataChanged
    {
        add
        {
            lock (_lockObject)
            {
                if (value != null)
                {
                    // Check if handler implements IEventHandler
                    if (value.Target is IEventHandler handler)
                    {
                        if (handler.IsEnabled)
                        {
                            Console.WriteLine($"Adding enabled handler: {handler.HandlerName}");
                            _dataChanged += value;
                            _registeredHandlers.Add(handler);
                        }
                        else
                        {
                            Console.WriteLine($"Skipping disabled handler: {handler.HandlerName}");
                        }
                    }
                    else
                    {
                        Console.WriteLine($"Adding standard handler: {value.Method.Name}");
                        _dataChanged += value;
                    }
                }
            }
        }
        remove
        {
            lock (_lockObject)
            {
                if (value != null)
                {
                    if (value.Target is IEventHandler handler)
                    {
                        Console.WriteLine($"Removing filtered handler: {handler.HandlerName}");
                        _registeredHandlers.Remove(handler);
                    }
                    else
                    {
                        Console.WriteLine($"Removing standard handler: {value.Method.Name}");
                    }
                }
                _dataChanged -= value;
            }
        }
    }

    public void ChangeData(string newData)
    {
        Console.WriteLine($"Changing data to: {newData}");
        _dataChanged?.Invoke(this, newData);
    }

    public void ListRegisteredHandlers()
    {
        lock (_lockObject)
        {
            Console.WriteLine("Registered handlers:");
            foreach (var handler in _registeredHandlers)
            {
                Console.WriteLine($"  - {handler.HandlerName} (Enabled: {handler.IsEnabled})");
            }
        }
    }
}

class Program
{
    static void Main()
    {
        FilteredEventPublisher publisher = new FilteredEventPublisher();
        
        // Create filtered handlers
        FilteredEventHandler handler1 = new FilteredEventHandler("Handler1", true);
        FilteredEventHandler handler2 = new FilteredEventHandler("Handler2", false);
        FilteredEventHandler handler3 = new FilteredEventHandler("Handler3", true);

        Console.WriteLine("=== Event Filtering ===");
        
        // Subscribe to events
        publisher.DataChanged += handler1.OnDataChanged;
        publisher.DataChanged += handler2.OnDataChanged; // Disabled
        publisher.DataChanged += handler3.OnDataChanged;

        // List registered handlers
        publisher.ListRegisteredHandlers();

        // Trigger event
        publisher.ChangeData("Filtered data");

        Console.WriteLine("\n=== Enabling Disabled Handler ===");
        
        // Enable disabled handler
        handler2.IsEnabled = true;
        publisher.DataChanged += handler2.OnDataChanged;

        // List registered handlers again
        publisher.ListRegisteredHandlers();

        // Trigger event
        publisher.ChangeData("Updated data");
    }
}
```

**Output:**
```
=== Event Filtering ===
Adding enabled handler: Handler1
Skipping disabled handler: Handler2
Adding enabled handler: Handler3
Registered handlers:
  - Handler1 (Enabled: True)
  - Handler3 (Enabled: True)
Changing data to: Filtered data
  [Handler1] Data changed: Filtered data
  [Handler3] Data changed: Filtered data

=== Enabling Disabled Handler ===
Adding enabled handler: Handler2
Registered handlers:
  - Handler1 (Enabled: True)
  - Handler3 (Enabled: True)
  - Handler2 (Enabled: True)
Changing data to: Updated data
  [Handler1] Data changed: Updated data
  [Handler3] Data changed: Updated data
  [Handler2] Data changed: Updated data
```

### Example 4: Event Validation with Custom Accessors

```csharp
using System;
using System.Collections.Generic;

// Event handler validator
class EventHandlerValidator
{
    public static bool ValidateHandler(EventHandler<string> handler)
    {
        if (handler == null)
        {
            Console.WriteLine("Validation failed: Handler is null");
            return false;
        }

        if (handler.Method == null)
        {
            Console.WriteLine("Validation failed: Handler method is null");
            return false;
        }

        if (handler.Method.ReturnType != typeof(void))
        {
            Console.WriteLine("Validation failed: Handler must return void");
            return false;
        }

        var parameters = handler.Method.GetParameters();
        if (parameters.Length != 2)
        {
            Console.WriteLine("Validation failed: Handler must have exactly 2 parameters");
            return false;
        }

        if (parameters[0].ParameterType != typeof(object))
        {
            Console.WriteLine("Validation failed: First parameter must be object");
            return false;
        }

        if (parameters[1].ParameterType != typeof(string))
        {
            Console.WriteLine("Validation failed: Second parameter must be string");
            return false;
        }

        Console.WriteLine($"Validation passed: {handler.Method.Name}");
        return true;
    }
}

// Class with validated event accessors
class ValidatedEventPublisher
{
    private EventHandler<string> _dataChanged;
    private readonly List<EventHandler<string>> _validatedHandlers = new List<EventHandler<string>>();
    private readonly object _lockObject = new object();

    // Custom event accessors with validation
    public event EventHandler<string> DataChanged
    {
        add
        {
            lock (_lockObject)
            {
                if (value != null)
                {
                    if (EventHandlerValidator.ValidateHandler(value))
                    {
                        _dataChanged += value;
                        _validatedHandlers.Add(value);
                        Console.WriteLine($"Handler added successfully: {value.Method.Name}");
                    }
                    else
                    {
                        Console.WriteLine($"Handler validation failed: {value.Method.Name}");
                    }
                }
            }
        }
        remove
        {
            lock (_lockObject)
            {
                if (value != null)
                {
                    _dataChanged -= value;
                    _validatedHandlers.Remove(value);
                    Console.WriteLine($"Handler removed: {value.Method.Name}");
                }
            }
        }
    }

    public void ChangeData(string newData)
    {
        Console.WriteLine($"Changing data to: {newData}");
        _dataChanged?.Invoke(this, newData);
    }

    public void ListValidatedHandlers()
    {
        lock (_lockObject)
        {
            Console.WriteLine("Validated handlers:");
            foreach (var handler in _validatedHandlers)
            {
                Console.WriteLine($"  - {handler.Method.Name}");
            }
        }
    }
}

// Valid event handler
class ValidEventHandler
{
    public void OnDataChanged(object sender, string data)
    {
        Console.WriteLine($"  [VALID] Data changed: {data}");
    }
}

// Invalid event handler (wrong signature)
class InvalidEventHandler
{

{

        // Remove handlers
        publisher.DataChanged -= validHandler.OnDataChanged;
        publisher.DataChanged -= invalidHandler.OnDataChanged;

        // List validated handlers again

}
```

### Example 5: Custom Event Storage with Accessors

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

// Custom event storage
class EventStorage<T> where T : EventArgs
{
    private readonly List<EventHandler<T>> _handlers = new List<EventHandler<T>>();
    private readonly object _lockObject = new object();

    public void AddHandler(EventHandler<T> handler)
    {
        lock (_lockObject)
        {
            if (handler != null && !_handlers.Contains(handler))
            {
                _handlers.Add(handler);
                Console.WriteLine($"Added handler: {handler.Method.Name}");
            }
        }
    }

    public void RemoveHandler(EventHandler<T> handler)
    {
        lock (_lockObject)
        {
            if (handler != null)
            {
                _handlers.Remove(handler);
                Console.WriteLine($"Removed handler: {handler.Method.Name}");
            }
        }
    }

    public void InvokeHandlers(object sender, T args)
    {
        EventHandler<T>[] handlers;
        lock (_lockObject)
        {
            handlers = _handlers.ToArray();
        }

        foreach (var handler in handlers)
        {
            try
            {
                handler?.Invoke(sender, args);
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Error in handler {handler?.Method.Name}: {ex.Message}");
            }
        }
    }

    public int HandlerCount
    {
        get
        {
            lock (_lockObject)
            {
                return _handlers.Count;
            }
        }
    }

    public void ListHandlers()
    {
        lock (_lockObject)
        {
            Console.WriteLine($"Handlers ({_handlers.Count}):");
            foreach (var handler in _handlers)
            {
                Console.WriteLine($"  - {handler.Method.Name}");
            }
        }
    }
}

// Custom event arguments
class CustomEventArgs : EventArgs
{
    public string Message { get; }
    public DateTime Timestamp { get; }

    public CustomEventArgs(string message)
    {
        Message = message;
        Timestamp = DateTime.Now;
    }
}

// Class with custom event storage
class CustomStorageEventPublisher
{
    private readonly EventStorage<CustomEventArgs> _eventStorage;

    public CustomStorageEventPublisher()
    {
        _eventStorage = new EventStorage<CustomEventArgs>();
    }

    // Custom event accessors with custom storage
    public event EventHandler<CustomEventArgs> CustomEvent
    {
        add
        {
            _eventStorage.AddHandler(value);
        }
        remove
        {
            _eventStorage.RemoveHandler(value);
        }
    }

    public void TriggerCustomEvent(string message)
    {
        Console.WriteLine($"Triggering custom event: {message}");
        var args = new CustomEventArgs(message);
        _eventStorage.InvokeHandlers(this, args);
    }

    public void ListEventHandlers()
    {
        _eventStorage.ListHandlers();
    }

    public int EventHandlerCount => _eventStorage.HandlerCount;
}

// Event handler
class CustomEventHandler
{
    private string _name;

    public CustomEventHandler(string name)
    {
        _name = name;
    }

    public void OnCustomEvent(object sender, CustomEventArgs e)
    {
        Console.WriteLine($"  [{_name}] Custom event: {e.Message} at {e.Timestamp:HH:mm:ss}");
    }
}

class Program
{

    {
        CustomStorageEventPublisher publisher = new CustomStorageEventPublisher();
        CustomEventHandler handler1 = new CustomEventHandler("Handler1");
        CustomEventHandler handler2 = new CustomEventHandler("Handler2");
        CustomEventHandler handler3 = new CustomEventHandler("Handler3");

        Console.WriteLine("=== Custom Event Storage ===");

        publisher.CustomEvent += handler2.OnCustomEvent;
        publisher.CustomEvent += handler3.OnCustomEvent;

        // List handlers

        // Trigger event
        publisher.TriggerCustomEvent("Custom message 1");

        Console.WriteLine("\n=== Partial Unsubscription ===");
        
        // Remove one handler
        publisher.CustomEvent -= handler2.OnCustomEvent;

        // List handlers again
        publisher.ListEventHandlers();

        // Trigger event
        publisher.TriggerCustomEvent("Custom message 2");

        Console.WriteLine($"\nHandler count: {publisher.EventHandlerCount}");
    }
}
```

### Common Use Cases
- **Event Filtering**: Filter handlers based on criteria
- **Event Logging**: Log event subscription and unsubscription
- **Event Validation**: Validate handler signatures
