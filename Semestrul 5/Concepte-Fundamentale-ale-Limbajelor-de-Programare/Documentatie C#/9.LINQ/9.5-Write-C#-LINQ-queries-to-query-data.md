You'll learn how to combine different LINQ operators to create complex queries, handle various data scenarios, and write efficient, maintainable code. The examples cover common patterns and best practices for querying different types of data sources.

- **Query composition**: Building complex queries from simple parts
- **Real-world scenarios**: Practical examples for common data querying needs
- **Performance optimization**: Writing efficient queries
- **Error handling**: Dealing with null values and edge cases

    static void CustomerAnalysis(List<Customer> customers, List<Order> orders)

    static void ProductAnalysis(List<Product> products, List<OrderItem> orderItems)
    {
        Console.WriteLine("\n=== Product Analysis ===");
        
        // Best-selling products
        var bestSellingProducts = from p in products
                                 join oi in orderItems on p.Id equals oi.ProductId
                                 join o in orderItems.Select(oi => oi.OrderId).Distinct() on oi.OrderId equals o
                                 group oi by new { p.Id, p.Name, p.Category, p.Price } into g
                                 let totalQuantity = g.Sum(oi => oi.Quantity)
                                 let totalRevenue = g.Sum(oi => oi.Quantity * oi.UnitPrice)
                                 orderby totalQuantity descending
                                 select new
                                 {
                                     g.Key.Name,
                                     g.Key.Category,
                                     g.Key.Price,
                                     TotalQuantity = totalQuantity,
                                     TotalRevenue = totalRevenue
                                 };
        
        Console.WriteLine("Best-selling products:");
        foreach (var product in bestSellingProducts.Take(5))
        {
            Console.WriteLine($"  {product.Name} ({product.Category}): " +
                            $"{product.TotalQuantity} units, " +
                            $"${product.TotalRevenue:N2} revenue");
        }
        
        // Product performance by category
        var categoryPerformance = from p in products
                                 join oi in orderItems on p.Id equals oi.ProductId into productOrders
                                 let totalSold = productOrders.Sum(oi => oi.Quantity)
                                 let totalRevenue = productOrders.Sum(oi => oi.Quantity * oi.UnitPrice)
                                 group new { p, totalSold, totalRevenue } by p.Category into g
                                 let categoryTotalSold = g.Sum(x => x.totalSold)
                                 let categoryTotalRevenue = g.Sum(x => x.totalRevenue)
                                 let productCount = g.Count()
                                 orderby categoryTotalRevenue descending
                                 select new
                                 {
                                     Category = g.Key,
                                     ProductCount = productCount,
                                     TotalSold = categoryTotalSold,
                                     TotalRevenue = categoryTotalRevenue,
                                     AverageRevenuePerProduct = categoryTotalRevenue / productCount
                                 };
        
        Console.WriteLine("\nProduct performance by category:");
        foreach (var category in categoryPerformance)
        {
            Console.WriteLine($"  {category.Category}: " +
                            $"{category.ProductCount} products, " +
                            $"{category.TotalSold} units sold, " +
                            $"${category.TotalRevenue:N2} revenue " +
                            $"(Avg: ${category.AverageRevenuePerProduct:N2}/product)");
        }
    }
    
    static void SalesAnalysis(List<Order> orders, List<OrderItem> orderItems)
    {
        Console.WriteLine("\n=== Sales Analysis ===");
        
        // Monthly sales trend
        var monthlySales = from o in orders
                          where o.Status == "Completed"
                          group o by new { o.OrderDate.Year, o.OrderDate.Month } into g
                          let totalRevenue = g.Sum(o => o.TotalAmount)
                          let orderCount = g.Count()
                          orderby g.Key.Year, g.Key.Month
                          select new
                          {
                              Month = $"{g.Key.Year}-{g.Key.Month:D2}",
                              OrderCount = orderCount,
                              TotalRevenue = totalRevenue,
                              AverageOrderValue = totalRevenue / orderCount
                          };
        
        Console.WriteLine("Monthly sales trend:");
        foreach (var month in monthlySales)
        {
            Console.WriteLine($"  {month.Month}: " +
                            $"{month.OrderCount} orders, " +
                            $"${month.TotalRevenue:N2} revenue " +
                            $"(Avg: ${month.AverageOrderValue:N2})");
        }
        
        // Order value distribution
        var orderValueDistribution = from o in orders
                                    where o.Status == "Completed"
                                    let valueRange = o.TotalAmount switch
                                    {
                                        var amount when amount < 50 => "Low ($0-$50)",
                                        var amount when amount < 200 => "Medium ($50-$200)",
                                        var amount when amount < 500 => "High ($200-$500)",
                                        _ => "Very High ($500+)"
                                    }
                                    group o by valueRange into g
                                    orderby g.Key
                                    select new
                                    {
                                        ValueRange = g.Key,
                                        OrderCount = g.Count(),
                                        TotalRevenue = g.Sum(o => o.TotalAmount),
                                        Percentage = (double)g.Count() / orders.Count(o => o.Status == "Completed") * 100
                                    };
        
        Console.WriteLine("\nOrder value distribution:");
        foreach (var range in orderValueDistribution)
        {
            Console.WriteLine($"  {range.ValueRange}: " +
                            $"{range.OrderCount} orders ({range.Percentage:F1}%), " +
                            $"${range.TotalRevenue:N2} revenue");
        }
    }
    
    static void InventoryAnalysis(List<Product> products, List<OrderItem> orderItems)
    {
        Console.WriteLine("\n=== Inventory Analysis ===");
        
        // Low stock products
        var lowStockProducts = from p in products
                              join oi in orderItems on p.Id equals oi.ProductId into productOrders
                              let totalSold = productOrders.Sum(oi => oi.Quantity)
                              let currentStock = p.Stock - totalSold
                              where currentStock < 20
                              orderby currentStock
                              select new
                              {
                                  p.Name,
                                  p.Category,
                                  CurrentStock = currentStock,
                                  OriginalStock = p.Stock,
                                  TotalSold = totalSold,
                                  StockPercentage = (double)currentStock / p.Stock * 100
                              };
        
        Console.WriteLine("Low stock products (less than 20 units):");
        foreach (var product in lowStockProducts)
        {
            Console.WriteLine($"  {product.Name} ({product.Category}): " +
                            $"{product.CurrentStock} units remaining " +
                            $"({product.StockPercentage:F1}% of original {product.OriginalStock})");
        }
        
        // Inventory turnover analysis
        var inventoryTurnover = from p in products
                               join oi in orderItems on p.Id equals oi.ProductId into productOrders
                               let totalSold = productOrders.Sum(oi => oi.Quantity)
                               let turnoverRate = p.Stock > 0 ? (double)totalSold / p.Stock : 0
                               orderby turnoverRate descending
                               select new
                               {
                                   p.Name,
                                   p.Category,
                                   p.Stock,
                                   TotalSold = totalSold,
                                   TurnoverRate = turnoverRate,
                                   Status = turnoverRate switch
                                   {
                                       var rate when rate >= 1.0 => "High Turnover",
                                       var rate when rate >= 0.5 => "Medium Turnover",
                                       var rate when rate > 0 => "Low Turnover",
                                       _ => "No Sales"
                                   }
                               };
        
        Console.WriteLine("\nInventory turnover analysis:");
        foreach (var product in inventoryTurnover)
        {
            Console.WriteLine($"  {product.Name} ({product.Category}): " +
                            $"{product.TotalSold} sold, {product.Stock} in stock " +
                            $"(Rate: {product.TurnoverRate:F2}) - {product.Status}");
        }
    }

    static void EmployeePerformanceAnalysis(List<Employee> employees, List<Project> projects, List<TaskAssignment> taskAssignments)
    {
        Console.WriteLine("\n=== Employee Performance Analysis ===");
        
        // Employee workload analysis
        var employeeWorkload = from e in employees
                              join ta in taskAssignments on e.Id equals ta.EmployeeId into employeeTasks
                              let totalHours = employeeTasks.Sum(t => t.HoursWorked)
                              let completedTasks = employeeTasks.Count(t => t.Status == "Completed")
                              let inProgressTasks = employeeTasks.Count(t => t.Status == "In Progress")
                              let pendingTasks = employeeTasks.Count(t => t.Status == "Pending")
                              let totalTasks = employeeTasks.Count()
                              orderby totalHours descending
                              select new
                              {
                                  e.Name,
                                  e.Position,
                                  e.DepartmentId,
                                  TotalHours = totalHours,
                                  TotalTasks = totalTasks,
                                  CompletedTasks = completedTasks,
                                  InProgressTasks = inProgressTasks,
                                  PendingTasks = pendingTasks,
                                  CompletionRate = totalTasks > 0 ? (double)completedTasks / totalTasks * 100 : 0
                              };
        
        Console.WriteLine("Employee workload analysis:");
        foreach (var emp in employeeWorkload)
        {
            Console.WriteLine($"  {emp.Name} ({emp.Position}): " +
                            $"{emp.TotalHours} hours, {emp.TotalTasks} tasks " +
                            $"({emp.CompletedTasks} completed, {emp.InProgressTasks} in progress, {emp.PendingTasks} pending) " +
                            $"- {emp.CompletionRate:F1}% completion rate");
        }
        
        // Top performers by hours worked
        var topPerformers = employeeWorkload.Take(3);
        Console.WriteLine("\nTop performers by hours worked:");
        foreach (var performer in topPerformers)
        {
            Console.WriteLine($"  {performer.Name}: {performer.TotalHours} hours");
        }
        
        // Employee utilization by department
        var departmentUtilization = from emp in employeeWorkload
                                   group emp by emp.DepartmentId into g
                                   let avgHours = g.Average(e => e.TotalHours)
                                   let avgCompletionRate = g.Average(e => e.CompletionRate)
                                   let totalHours = g.Sum(e => e.TotalHours)
                                   orderby avgHours descending
                                   select new
                                   {
                                       DepartmentId = g.Key,
                                       EmployeeCount = g.Count(),
                                       AvgHours = avgHours,
                                       TotalHours = totalHours,
                                       AvgCompletionRate = avgCompletionRate
                                   };
        
        Console.WriteLine("\nDepartment utilization:");
        foreach (var dept in departmentUtilization)

    static void SkillsAnalysis(List<Employee> employees, List<Skill> skills)

                                from skill in e.Skills
                                group skill by skill into g
                                let skillInfo = skills.FirstOrDefault(s => s.Name == g.Key)
                                orderby g.Count() descending
                                select new
                                {
                                    Skill = g.Key,
                                    Count = g.Count(),
                                    Category = skillInfo?.Category ?? "Unknown",
                                    DemandLevel = skillInfo?.DemandLevel ?? "Unknown"
                                };
        
        Console.WriteLine("Skills distribution:");
        foreach (var skill in skillsDistribution)

            Console.WriteLine($"  {skill.Skill} ({skill.Category}): " +

                            $"Demand: {skill.DemandLevel}");

        // Skills gap analysis
        var skillsGap = from s in skills
                        let employeesWithSkill = employees.Count(e => e.Skills.Contains(s.Name))
                        let totalEmployees = employees.Count
                        let coveragePercentage = (double)employeesWithSkill / totalEmployees * 100
                        orderby coveragePercentage
                        select new
                        {
                            s.Name,
                            s.Category,
                            s.DemandLevel,
                            EmployeesWithSkill = employeesWithSkill,
                            CoveragePercentage = coveragePercentage,
                            Gap = totalEmployees - employeesWithSkill

        Console.WriteLine("\nSkills gap analysis:");
        foreach (var skill in skillsGap)
        {
            Console.WriteLine($"  {skill.Name} ({skill.Category}): " +
                            $"{skill.EmployeesWithSkill}/{employees.Count} employees " +
                            $"({skill.CoveragePercentage:F1}% coverage, " +
                            $"Gap: {skill.Gap}, Demand: {skill.DemandLevel})");
        }
        
        // Employee skills profile
        var employeeSkillsProfile = from e in employees

                                   let highDemandSkills = e.Skills.Where(skill => 
                                       skills.Any(s => s.Name == skill && s.DemandLevel == "High")).Count()
                                   let skillCategories = e.Skills.Select(skill =>

                                       e.Name,
                                       e.Position,
                                       SkillCount = skillCount,
                                       HighDemandSkills = highDemandSkills,
                                       SkillCategories = skillCategories.ToList()

                            $"{emp.HighDemandSkills} high-demand skills, " +

    static void DataTransformationPipelines(SampleData data)
    {
        Console.WriteLine("\n=== Data Transformation Pipelines ===");
        
        // Pipeline 1: Employee data enrichment
        var enrichedEmployees = data.Employees
            .Where(e => e.Salary > 70000) // Filter high earners
            .Select(e => new
            {
                e.Name,
                e.Department,
                e.Salary,
                SalaryCategory = e.Salary switch
                {
                    var s when s >= 90000 => "High",
                    var s when s >= 80000 => "Medium-High",
                    _ => "Medium"
                },
                Company = data.Companies.FirstOrDefault(c => c.Id == e.CompanyId)?.Name ?? "Unknown"
            })
            .OrderByDescending(e => e.Salary)
            .ToList();
        
        Console.WriteLine("Enriched employee data (high earners):");
        foreach (var emp in enrichedEmployees)
        {
            Console.WriteLine($"  {emp.Name} ({emp.Company}, {emp.Department}): " +
                            $"${emp.Salary:N0} - {emp.SalaryCategory}");
        }
        
        // Pipeline 2: Sales analysis with rolling averages
        var salesAnalysis = data.Sales
            .OrderBy(s => s.Date)
            .Select((sale, index) => new
            {
                sale.Date,
                sale.Amount,
                sale.Product,
                Company = data.Companies.FirstOrDefault(c => c.Id == sale.CompanyId)?.Name ?? "Unknown",
                RunningTotal = data.Sales
                    .Where(s => s.Date <= sale.Date)
                    .Sum(s => s.Amount),
                MonthlyAverage = data.Sales
                    .Where(s => s.Date.Year == sale.Date.Year && s.Date.Month == sale.Date.Month)
                    .Average(s => s.Amount)
            })
            .ToList();
        
        Console.WriteLine("\nSales analysis with running totals:");
        foreach (var sale in salesAnalysis)
        {
            Console.WriteLine($"  {sale.Date:MM/dd/yyyy} - {sale.Company}: " +
                            $"${sale.Amount:N0} (Running Total: ${sale.RunningTotal:N0}, " +
                            $"Monthly Avg: ${sale.MonthlyAverage:N0})");
        }
        
        // Pipeline 3: Complex data aggregation and ranking
        var companyRankings = data.Companies
            .Select(c => new
            {
                Company = c,
                Employees = data.Employees.Where(e => e.CompanyId == c.Id).ToList(),
                Sales = data.Sales.Where(s => s.CompanyId == c.Id).ToList()
            })
            .Select(c => new
            {
                c.Company.Name,
                c.Company.Industry,
                EmployeeCount = c.Employees.Count,
                TotalSalary = c.Employees.Sum(e => e.Salary),
                TotalSales = c.Sales.Sum(s => s.Amount),
                AvgSalary = c.Employees.Any() ? c.Employees.Average(e => e.Salary) : 0,
                AvgSaleAmount = c.Sales.Any() ? c.Sales.Average(s => s.Amount) : 0,
                RevenuePerEmployee = c.Employees.Any() ? c.Sales.Sum(s => s.Amount) / c.Employees.Count : 0
            })
            .OrderByDescending(c => c.TotalSales)
            .Select((c, index) => new
            {
                Rank = index + 1,
                c.Name,
                c.Industry,
                c.EmployeeCount,
                c.TotalSales,
                c.AvgSalary,
                c.RevenuePerEmployee,
                PerformanceScore = (c.TotalSales / 1000.0) + (c.AvgSalary / 1000.0) + (c.RevenuePerEmployee / 1000.0)
            })
            .ToList();
        
        Console.WriteLine("\nCompany rankings:");
        foreach (var company in companyRankings)
        {
            Console.WriteLine($"  #{company.Rank} {company.Name} ({company.Industry}): " +
                            $"${company.TotalSales:N0} sales, " +
                            $"${company.AvgSalary:N0} avg salary, " +
                            $"${company.RevenuePerEmployee:N0}/employee, " +
                            $"Score: {company.PerformanceScore:F2}");
        }
    }

## Best Practices for Writing LINQ Queries

### 1. Query Structure
- Use meaningful variable names
- Break complex queries into smaller parts
- Use query syntax for complex operations
- Use method syntax for simple operations

### 2. Performance Optimization
- Use `Where` before `Select` to reduce data early
- Avoid multiple enumerations of the same query
- Use `AsParallel()` for CPU-intensive operations
- Consider using `AsQueryable()` for database queries

### 3. Error Handling
- Handle null values with null-conditional operators
- Use `FirstOrDefault()` instead of `First()` when appropriate
- Validate data before querying

### 4. Readability
- Use descriptive variable names
- Add comments for complex logic
- Format queries for readability
- Use consistent indentation

### 5. Testing
- Test with edge cases (empty collections, null values)
- Verify query results with expected outcomes
- Use unit tests for complex queries

## Common Query Patterns

1. **Filtering and Projection**: `Where` + `Select`
2. **Grouping and Aggregation**: `GroupBy` + aggregation functions
3. **Sorting and Paging**: `OrderBy` + `Skip` + `Take`
4. **Joining Data**: `Join` or `GroupJoin`
5. **Data Transformation**: `Select` with anonymous types
6. **Conditional Logic**: `Where` with complex conditions
7. **Aggregation with Grouping**: `GroupBy` + `Select` with aggregations
8. **Hierarchical Queries**: Nested `GroupBy` operations
